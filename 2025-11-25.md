### 顺序表核心算法实现
```c
// 顺序表插入操作
Status ListInsert(SqList *p, ElemType *elem, int index) {
    if(index < 1 || index > p->length + 1) return ERROR;
    if(p->length == MAXSIZE) return ERROR;
    
    for(int i = p->length - 1; i >= index - 1; i--) {
        p->elem[i + 1] = p->elem[i];
    }
    p->elem[index - 1] = *elem;
    p->length++;
    return OK;
}

// 顺序表删除操作  
Status ListDelete(SqList *p, int i, ElemType *elem) {
    if(i < 1 || i > p->length) return ERROR;
    *elem = p->elem[i - 1];
    
    for(int j = i; j <= p->length - 1; j++) {
        p->elem[j - 1] = p->elem[j];
    }
    p->length--;
    return OK;
}
```
